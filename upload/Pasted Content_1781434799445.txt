const setActiveKey = async (keyId: string) => {
  // Deactivate all first
  await supabaseAdmin
    .from('gemini_keys')
    .update({ is_active: false, status: 'standby' })
    .neq('id', keyId)

  // Activate selected
  await supabaseAdmin
    .from('gemini_keys')
    .update({ is_active: true, status: 'active' })
    .eq('id', keyId)

  showToast('✅ Key activated!', 'success')
  fetchKeys()
  
  // Log activity
  await supabaseAdmin
    .from('admin_activity_log')
    .insert([{
      action: 'SWITCH_GEMINI_KEY',
      target: keyId,
      details: { manual: true }
    }])
}

── AUTO ROTATE LOGIC ──
const triggerAutoRotate = async () => {
  const active = keys.find(k => k.is_active)
  if (!active) return
  
  // Check if quota exceeded threshold
  if (active.quota_used < rotateThreshold) return
  
  // Find next standby key
  const next = keys.find(
    k => k.status === 'standby' && 
    k.id !== active.id
  )
  if (!next) {
    showToast(
      '⚠️ No standby keys available!', 'error')
    return
  }
  
  // Rotate
  await supabaseAdmin
    .from('gemini_keys')
    .update({ 
      status: 'cooling', 
      is_active: false 
    })
    .eq('id', active.id)
  
  await supabaseAdmin
    .from('gemini_keys')
    .update({ 
      status: 'active', 
      is_active: true 
    })
    .eq('id', next.id)
  
  showToast(
    `🔄 Auto-rotated to ${next.key_name}`, 
    'success')
  fetchKeys()
}

── ADD KEY ──
const addKey = async () => {
  if (!newKeyName.trim() || !newKeyValue.trim()) {
    showToast('Fill all fields', 'error')
    return
  }
  if (keys.length >= 10) {
    showToast('Maximum 10 keys allowed', 'error')
    return
  }
  
  const { error } = await supabaseAdmin
    .from('gemini_keys')
    .insert([{
      key_name: newKeyName.trim(),
      api_key: newKeyValue.trim(),
      status: keys.length === 0 
        ? 'active' : 'standby',
      is_active: keys.length === 0,
      sort_order: keys.length,
    }])
  
  if (!error) {
    showToast('✅ Key added!', 'success')
    setShowAddModal(false)
    setNewKeyName('')
    setNewKeyValue('')
    fetchKeys()
  }
}

── DELETE KEY ──
const deleteKey = async (keyId: string) => {
  const key = keys.find(k => k.id === keyId)
  if (key?.is_active) {
    showToast(
      'Cannot delete active key. Switch first.', 
      'error')
    return
  }
  await supabaseAdmin
    .from('gemini_keys')
    .delete()
    .eq('id', keyId)
  showToast('Key deleted', 'info')
  fetchKeys()
}

  const { data } = await supabaseAdmin
    .from('api_vault')
    .select('*')
    .order('service')
  
  const grouped = (data || []).reduce((acc, item) => {

export async function POST(req: NextRequest) {
  const { message, history, apiKey, sessionId } = 
    await req.json()
  
  const key = apiKey || process.env.GEMINI_API_KEY
  if (!key) return NextResponse.json(
    { reply: '❌ No API key configured' })
  
  try {
    const genAI = new GoogleGenerativeAI(key)
    const model = genAI.getGenerativeModel({
      model: 'gemini-1.5-flash',
      systemInstruction: PLAY_NEXA_CONTEXT
    })
    
    // Build conversation history
    const chatHistory = history
      .slice(0, -1) // exclude last user msg
      .map((m: Message) => ({
        role: m.role === 'user' ? 'user' : 'model',
        parts: [{ text: m.content }]
      }))
    
    const chat = model.startChat({
      history: chatHistory
    })
    
    const result = await chat.sendMessage(message)
    const reply = result.response.text()
    
    return NextResponse.json({ reply })
  } catch (err: any) {
    return NextResponse.json({
      reply: `❌ API Error: ${err.message}`
    })

import { NextRequest, NextResponse } 
  from 'next/server'
import { supabaseAdmin } from 
  '@/lib/supabaseAdmin'
import { GoogleGenerativeAI } 
  from '@google/generative-ai'

export async function POST(req: NextRequest) {
  const { category, description, userId } = 
    await req.json()
  
  let priority = 'medium'
  let aiSummary = ''
  let aiVerified = false
  let isDuplicate = false
  let duplicateOf = null
  
  try {
    // Get active Gemini key
    const { data: activeKey } = await supabaseAdmin
      .from('gemini_keys')
      .select('api_key')
      .eq('is_active', true)
      .single()
    
    const apiKey = activeKey?.api_key || 
      process.env.GEMINI_API_KEY
    
    if (apiKey) {
      const genAI = new GoogleGenerativeAI(apiKey)
      const model = genAI.getGenerativeModel({
        model: 'gemini-1.5-flash'
      })
      
      const prompt = `
Analyze this user feedback for Play Nexa app:
Category: ${category}
Description: "${description}"

Respond ONLY in JSON:
{
  "is_real_issue": true/false,
  "priority": "high"/"medium"/"low",
  "summary": "one line summary in Bengali",
  "is_spam": true/false
}

Priority rules:
- high: app crash, nothing loads, login broken
- medium: feature not working properly
- low: suggestion, minor issue, cosmetic
`
      const result = await model.generateContent(prompt)
      const text = result.response.text()
      const json = text.match(/\{[\s\S]*?\}/)
      
      if (json) {
        const parsed = JSON.parse(json[0])
        priority = parsed.priority || 'medium'
        aiSummary = parsed.summary || ''
        aiVerified = parsed.is_real_issue || false
        
        if (parsed.is_spam) {
          return NextResponse.json({ 
            success: true,
            message: 'Thank you for your feedback!'
          })
        }
      }
    }
    
    // Check for duplicates
    const { data: recent } = await supabaseAdmin
      .from('user_feedback')
      .select('id, description')
      .eq('category', category)
      .eq('status', 'open')
      .gte('created_at', 
        new Date(Date.now() - 86400000).toISOString())
      .limit(10)
    
    if (recent && recent.length > 0) {
      const similar = recent.find(f =>
        f.description.toLowerCase().includes(
          description.toLowerCase().slice(0, 20)
        )
      )
      if (similar) {
        isDuplicate = true
        duplicateOf = similar.id
        // Increment same_issue_count
        await supabaseAdmin
          .from('user_feedback')
          .update({ 
            same_issue_count: 
              supabaseAdmin.raw(
                'same_issue_count + 1') 
          })
          .eq('id', similar.id)
      }
    }
    
  } catch {}
  
  // Save feedback
  await supabaseAdmin
    .from('user_feedback')
    .insert([{
      user_id: userId,
      category,
      description,
      ai_verified: aiVerified,
      is_duplicate: isDuplicate,
      duplicate_of: duplicateOf,
      priority,
      ai_summary: aiSummary,
      status: 'open',
    }])
  
  // Check for spike (5+ same category in 1 hour)
  const { count } = await supabaseAdmin
    .from('user_feedback')
    .select('*', { count: 'exact', head: true })
    .eq('category', category)
    .eq('priority', 'high')
    .gte('created_at',
      new Date(Date.now() - 3600000).toISOString())
  
  if ((count || 0) >= 5) {
    // Alert: spike detected
    await supabaseAdmin
      .from('admin_activity_log')
      .insert([{
        action: 'FEEDBACK_SPIKE',
        target: category,
        details: { 
          count, 
          message: `${count} high priority ${category} reports in 1 hour!`
        }
      }])
  }
  
  return NextResponse.json({ 
    success: true,
    message: 'Thank you for your feedback!'
  })
}

export async function POST() {
  const today = new Date()
  today.setHours(0,0,0,0)
  const yesterday = new Date(today)
  yesterday.setDate(yesterday.getDate() - 1)
  
  // Get yesterday's feedbacks
  const { data: feedbacks } = await supabaseAdmin
    .from('user_feedback')
    .select('*')
    .gte('created_at', yesterday.toISOString())
    .lt('created_at', today.toISOString())
  
  if (!feedbacks?.length) {
    return NextResponse.json({ 
      message: 'No feedback today' })
  }
  
  const high = feedbacks.filter(
    f => f.priority === 'high')
  const medium = feedbacks.filter(
    f => f.priority === 'medium')
  const low = feedbacks.filter(
    f => f.priority === 'low')
  
  // Group by category
  const byCategory = feedbacks.reduce((acc, f) => {
    acc[f.category] = (acc[f.category] || 0) + 1
    return acc
  }, {} as Record<string, number>)
  
  const topIssues = Object.entries(byCategory)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 5)
    .map(([category, count]) => ({
      category,
      count,
      sample: feedbacks
        .filter(f => f.category === category)[0]
        ?.ai_summary || ''
    }))
  
  // Save report
  await supabaseAdmin
    .from('admin_reports')
    .upsert([{
      report_date: yesterday.toISOString()
        .split('T')[0],
      total_feedback: feedbacks.length,
      high_count: high.length,
      medium_count: medium.length,
      low_count: low.length,
      top_issues: topIssues,
      full_report: {
        feedbacks: feedbacks.map(f => ({
          category: f.category,
          description: f.description,
          priority: f.priority,
          ai_summary: f.ai_summary,
        }))
      }
    }], { onConflict: 'report_date' })
  
  return NextResponse.json({ success: true })
}
