'use client'
import { useState, useEffect } from 'react'
import { supabaseAdmin } from '@/lib/supabaseAdmin'

interface Feedback {
  id: string
  category: string
  description: string
  ai_summary: string
  priority: 'high' | 'medium' | 'low'
  status: 'open' | 'in_progress' | 'resolved'
  is_duplicate: boolean
  same_issue_count: number
  created_at: string
  ai_verified: boolean
}

interface DailyReport {
  id: string
  report_date: string
  total_feedback: number
  high_count: number
  medium_count: number
  low_count: number
  top_issues: Array<{
    category: string
    count: number
    sample: string
  }>
}

export default function FeedbackPage() {
  const [feedbacks, setFeedbacks] = useState<
    Feedback[]>([])
  const [stats, setStats] = useState({
    total: 0, high: 0, medium: 0, 
    low: 0, resolved: 0
  })
  const [filters, setFilters] = useState({
    priority: 'all',
    status: 'all',
    category: 'all',
  })
  const [isLoading, setIsLoading] = useState(true)
  const [selectedFeedback, setSelectedFeedback] = 
    useState<Feedback | null>(null)
  const [fixPrompt, setFixPrompt] = useState('')
  const [showFixModal, setShowFixModal] = 
    useState(false)
  const [loadingFix, setLoadingFix] = useState(false)
  const [reports, setReports] = useState<
    DailyReport[]>([])
  const [activeView, setActiveView] = useState<
    'feedback' | 'reports'>('feedback')
  const [spikes, setSpikes] = useState<
    Array<{category: string; count: number}>>([])

  // Load on mount + when filters change
  useEffect(() => {
    fetchFeedbacks()
    fetchReports()
    detectSpikes()
  }, [filters])

  const fetchFeedbacks = async () => {
    setIsLoading(true)
    let query = supabaseAdmin
      .from('user_feedback')
      .select('*')
      .order('created_at', { ascending: false })
      .limit(100)

    if (filters.priority !== 'all')
      query = query.eq('priority', filters.priority)
    if (filters.status !== 'all')
      query = query.eq('status', filters.status)
    if (filters.category !== 'all')
      query = query.eq('category', filters.category)

    const { data } = await query
    const all = data || []
    setFeedbacks(all)

    setStats({
      total: all.length,
      high: all.filter(
        f => f.priority === 'high').length,
      medium: all.filter(
        f => f.priority === 'medium').length,
      low: all.filter(
        f => f.priority === 'low').length,
      resolved: all.filter(
        f => f.status === 'resolved').length,
    })
    setIsLoading(false)
  }

  const fetchReports = async () => {
    const { data } = await supabaseAdmin
      .from('admin_reports')
      .select('*')
      .order('report_date', { ascending: false })
      .limit(7)
    setReports(data || [])
  }

  const detectSpikes = async () => {
    const oneHourAgo = new Date(
      Date.now() - 3600000).toISOString()
    
    const { data } = await supabaseAdmin
      .from('user_feedback')
      .select('category')
      .eq('priority', 'high')
      .gte('created_at', oneHourAgo)

    if (!data) return
    
    const grouped = data.reduce((acc, f) => {
      acc[f.category] = (acc[f.category] || 0) + 1
      return acc
    }, {} as Record<string, number>)
    
    const detected = Object.entries(grouped)
      .filter(([, count]) => count >= 3)
      .map(([category, count]) => ({ 
        category, count }))
    
    setSpikes(detected)
  }

  const updateStatus = async (
    id: string,
    status: string
  ) => {
    await supabaseAdmin
      .from('user_feedback')
      .update({ status })
      .eq('id', id)
    
    setFeedbacks(prev => prev.map(f =>
      f.id === id ? { ...f, status: status as any } : f
    ))
  }

  const getFixPrompt = async (fb: Feedback) => {
    setSelectedFeedback(fb)
    setLoadingFix(true)
    setShowFixModal(true)
    
    try {
      const res = await fetch('/api/admin/chat', {
        method: 'POST',
        headers: { 
          'Content-Type': 'application/json' },
        body: JSON.stringify({
          message: `Play Nexa app এ এই user 
সমস্যার জন্য Z.ai তে দেওয়ার
detailed fix prompt দাও:

Category: ${fb.category}
Priority: ${fb.priority}
Issue: ${fb.description}
AI Summary: ${fb.ai_summary}
Same issue reported by: 
  ${fb.same_issue_count} users

Z.ai prompt format এ লিখো।
STRICT RULES section সহ।`,
          history: [],
        })
      })
      const data = await res.json()
      setFixPrompt(data.reply || '')
    } catch {
      setFixPrompt('Could not generate prompt. 
        Check API key.')
    } finally {
      setLoadingFix(false)
    }
  }

  const PRIORITY_STYLE = {
    high: 'bg-red-900/30 text-red-400 
      border-red-800/50',
    medium: 'bg-yellow-900/30 text-yellow-400
      border-yellow-800/50',
    low: 'bg-green-900/30 text-green-400
      border-green-800/50',
  }

  const STATUS_STYLE = {
    open: 'bg-blue-900/30 text-blue-400',
    in_progress: 'bg-purple-900/30 text-purple-400',
    resolved: 'bg-green-900/30 text-green-400',
  }

  return (
    <div className="p-4 pb-24 min-h-screen">

      {/* Spike Alerts */}
      {spikes.length > 0 && (
        <div className="mb-4 space-y-2">
          {spikes.map(spike => (
            <div key={spike.category}
              className="bg-red-900/20 border
                border-red-700/50 rounded-xl
                px-4 py-3 flex items-center gap-3">
              <span className="text-red-400 text-xl">
                🚨
              </span>
              <div className="flex-1">
                <p className="text-red-400 font-semibold
                  text-sm">
                  Spike Detected!
                </p>
                <p className="text-red-300 text-xs">
                  {spike.count} high priority 
                  {spike.category} reports 
                  in last hour
                </p>
              </div>
            </div>
          ))}
        </div>
      )}

      {/* Page header */}
      <div className="flex items-center
        justify-between mb-6">
        <h1 className="text-white font-bold text-xl">
          📣 User Feedback
        </h1>
        <button
          onClick={fetchFeedbacks}
          className="w-10 h-10 bg-[#1A1A1A]
            rounded-xl flex items-center
            justify-center text-[#9CA3AF]
            active:bg-[#2D2D2D]">
          ↻
        </button>
      </div>

      {/* View Toggle */}
      <div className="flex gap-2 mb-6">
        <button
          onClick={() => setActiveView('feedback')}
          className={`flex-1 py-2.5 rounded-xl
            text-sm font-semibold min-h-[44px]
            ${activeView === 'feedback'
              ? 'bg-[#7C3AED] text-white'
              : 'bg-[#1A1A1A] text-[#9CA3AF]'}`}>
          📣 Live Feedback
        </button>
        <button
          onClick={() => setActiveView('reports')}
          className={`flex-1 py-2.5 rounded-xl
            text-sm font-semibold min-h-[44px]
            ${activeView === 'reports'
              ? 'bg-[#7C3AED] text-white'
              : 'bg-[#1A1A1A] text-[#9CA3AF]'}`}>
          📊 Daily Reports
        </button>
      </div>

      {activeView === 'feedback' ? (
        <>
          {/* Stats Cards */}
          <div className="grid grid-cols-2 gap-3
            mb-6">
            {[
              { label: 'Total', value: stats.total,
                color: '#7C3AED', emoji: '📊' },
              { label: '🔴 High', value: stats.high,
                color: '#EF4444', emoji: '' },
              { label: '🟡 Medium', 
                value: stats.medium,
                color: '#F59E0B', emoji: '' },
              { label: '✅ Resolved',
                value: stats.resolved,
                color: '#22C55E', emoji: '' },
            ].map(card => (
              <div key={card.label}
                className="bg-[#0F0F0F] border
                  border-[#1A1A1A] rounded-2xl p-4">
                <p className="text-[#9CA3AF] text-xs
                  mb-1">{card.label}</p>
                <p className="font-bold text-2xl"
                  style={{ color: card.color }}>
                  {card.value}
                </p>
              </div>
            ))}
          </div>

          {/* Filters */}
          <div className="space-y-3 mb-6">
            {/* Priority filter */}
            <div className="flex gap-2 overflow-x-auto
              scrollbar-hide pb-1">
              {['all','high','medium','low'].map(p => (
                <button
                  key={p}
                  onClick={() => setFilters(
                    prev => ({...prev, priority: p}))}
                  className={`px-4 py-2 rounded-full
                    text-xs font-medium whitespace-nowrap
                    min-h-[36px] flex-shrink-0
                    ${filters.priority === p
                      ? 'bg-[#7C3AED] text-white'
                      : 'bg-[#1A1A1A] text-[#9CA3AF]'
                    }`}>
                  {p === 'all' ? 'All Priority'
                    : p === 'high' ? '🔴 High'
                    : p === 'medium' ? '🟡 Medium'
                    : '✅ Low'}
                </button>

                  key={s}

                    prev => ({...prev, status: s}))}
                  className={`px-4 py-2 rounded-full
                    text-xs font-medium whitespace-nowrap
                    min-h-[36px] flex-shrink-0

              </div>
            ))
          )}
        </div>
      )}

      {/* Fix Prompt Modal */}
      {showFixModal && (
        <>
          <div
            className="fixed inset-0 z-[65]
              bg-black/80"
            onClick={() => {
              setShowFixModal(false)
              setFixPrompt('')

          />

            <div className="flex items-center
              justify-between px-5 py-4
              border-b border-[#1A1A1A]
              flex-shrink-0">
              <p className="text-white font-bold">
                🔧 Fix Prompt
              </p>
              <div className="flex gap-2">
                {fixPrompt && (
                  <button
                    onClick={() => {
                      navigator.clipboard.writeText(
                        fixPrompt)
                    }}
                    className="px-3 py-1.5
                      bg-[#7C3AED]/20 rounded-lg
                      text-[#A78BFA] text-xs
                      active:opacity-80">
                    📋 Copy
                  </button>
                )}
                <button
                  onClick={() => {
                    setShowFixModal(false)
                    setFixPrompt('')
                  }}
                  className="w-8 h-8 bg-[#1A1A1A]
                    rounded-lg text-[#9CA3AF] flex
                    items-center justify-center">
                  ✕
                </button>
              </div>
            </div>

            {/* Modal content */}
            <div className="flex-1 overflow-y-auto
              p-5">
              {loadingFix ? (
                <div className="flex flex-col
                  items-center justify-center h-full
                  gap-4">
                  <div className="w-10 h-10 border-2
                    border-[#7C3AED]
                    border-t-transparent
                    rounded-full animate-spin"/>
                  <p className="text-[#9CA3AF] text-sm">
                    AI generating fix prompt...
                  </p>
                </div>
              ) : (
                <pre className="text-white text-xs
                  whitespace-pre-wrap
                  font-mono leading-relaxed">
                  {fixPrompt}
                </pre>
              )}
            </div>
          </div>
        </>
      )}
    </div>
  )
}

import { NextRequest, NextResponse } 
  from 'next/server'
import { supabaseAdmin } from 
  '@/lib/supabaseAdmin'
import { GoogleGenerativeAI } 
  from '@google/generative-ai'

export async function POST(req: NextRequest) {
  // Security check for CRON
  const cronSecret = req.headers.get(
    'x-cron-secret')
  const envSecret = process.env.CRON_SECRET
  
  if (envSecret && cronSecret !== envSecret) {
    // Allow manual trigger from admin
    // (no secret needed from admin panel)
    const referer = req.headers.get('referer')
    if (!referer?.includes('/admin')) {
      return NextResponse.json(
        { error: 'Unauthorized' }, 
        { status: 401 })
    }
  }

  const today = new Date()
  today.setHours(0, 0, 0, 0)
  const yesterday = new Date(today)
  yesterday.setDate(yesterday.getDate() - 1)

  // Get all feedback from yesterday
  const { data: feedbacks } = await supabaseAdmin
    .from('user_feedback')
    .select('*')
    .gte('created_at', yesterday.toISOString())
    .lt('created_at', today.toISOString())
    .order('priority')

  const all = feedbacks || []

  if (all.length === 0) {
    return NextResponse.json({
      message: 'No feedback yesterday',
      report_date: yesterday.toISOString()
        .split('T')[0],
    })
  }

  const high = all.filter(
    f => f.priority === 'high')
  const medium = all.filter(
    f => f.priority === 'medium')
  const low = all.filter(
    f => f.priority === 'low')

  // Group by category
  const byCategory = all.reduce((acc, f) => {
    acc[f.category] = (acc[f.category] || 0) + 1
    return acc
  }, {} as Record<string, number>)

  const topIssues = Object.entries(byCategory)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 5)
    .map(([category, count]) => {
      const sample = all.find(
        f => f.category === category)
      return {
        category,
        count,
        sample: sample?.ai_summary || 
          sample?.description?.slice(0, 60) || '',
      }
    })

  // Generate AI summary of the day
  let aiDailySummary = ''
  try {
    const { data: activeKey } = await supabaseAdmin
      .from('gemini_keys')
      .select('api_key')
      .eq('is_active', true)
      .single()

    const apiKey = activeKey?.api_key || 
      process.env.GEMINI_API_KEY

    if (apiKey) {
      const genAI = new GoogleGenerativeAI(apiKey)
      const model = genAI.getGenerativeModel({
        model: 'gemini-1.5-flash'
      })

      const result = await model.generateContent(`
Play Nexa app এর আজকের user feedback summary:

Total: ${all.length} reports
High Priority: ${high.length}
Medium Priority: ${medium.length}
Low Priority: ${low.length}

Top Issues: ${topIssues.map(
  i => `${i.category}: ${i.count}x`
).join(', ')}

3-4 line এ বাংলায় summary দাও।
কোন সমস্যা জরুরি সেটা highlight করো।
`)
      aiDailySummary = result.response.text()
    }
  } catch {}

  // Save report
  const { error } = await supabaseAdmin
    .from('admin_reports')
    .upsert([{
      report_date: yesterday.toISOString()
        .split('T')[0],
      total_feedback: all.length,
      high_count: high.length,
      medium_count: medium.length,
      low_count: low.length,
      top_issues: topIssues,
      full_report: {
        ai_summary: aiDailySummary,
        by_category: byCategory,
        high_issues: high.map(f => ({
          category: f.category,
          description: f.description
            .slice(0, 100),
          ai_summary: f.ai_summary,
        })),
      },
      generated_at: new Date().toISOString(),
    }], { onConflict: 'report_date' })

  if (error) {
    return NextResponse.json(
      { error: error.message }, 
      { status: 500 })
  }

  return NextResponse.json({
    success: true,
    report_date: yesterday.toISOString()
      .split('T')[0],
    total: all.length,
    high: high.length,
    ai_summary: aiDailySummary,
  })
}
