  youtubeId: string

type SizeMode = 'fit' | 'stretch' | 'zoom'

export default function VideoPlayer({
  youtubeId, title, onAudioOnlyToggle

  const [isLandscape, setIsLandscape] = useState(false)
  const [sizeMode, setSizeMode] = useState<
    SizeMode>('fit')
  const [showSizeMenu, setShowSizeMenu] = useState(false)
  const [isLocked, setIsLocked] = useState(false)
  const [showLockHint, setShowLockHint] = useState(false)
  const [isAudioOnly, setIsAudioOnly] = useState(false)
  const [controlsVisible, setControlsVisible] = 
    useState(true)
  const [seekFeedback, setSeekFeedback] = useState<
    { side: 'left' | 'right'; show: boolean }>(
    { side: 'left', show: false })
  const playerRef = useRef<HTMLDivElement>(null)
  const hideTimerRef = useRef<NodeJS.Timeout>()

  // ── LOCK SCREEN — PlayIt-style ──
  // When locked: ALL taps/gestures do nothing
  // except long-press to unlock
  const handleContainerTap = useCallback(() => {
    if (isLocked) {
      setShowLockHint(true)
      setTimeout(() => setShowLockHint(false), 1500)
      return
    }
    setControlsVisible(v => !v)
    resetHideTimer()
  }, [isLocked])

  const resetHideTimer = () => {
    if (hideTimerRef.current) 
      clearTimeout(hideTimerRef.current)
    hideTimerRef.current = setTimeout(() => {
      setControlsVisible(false)
    }, 3000)
  }

  let unlockPressTimer: NodeJS.Timeout
  const handleLockIconPressStart = () => {
    unlockPressTimer = setTimeout(() => {
      setIsLocked(false)
    }, 1000)
  }
  const handleLockIconPressEnd = () => {
    clearTimeout(unlockPressTimer)
  }

  // ── DOUBLE-TAP SEEK — width-relative zones ──
  // Works correctly in portrait AND landscape
  // because zones are % of CURRENT container width
  const lastTapRef = useRef<{ 
    time: number; x: number }>({ time: 0, x: 0 })

  const handleZoneTap = (
    e: React.MouseEvent<HTMLDivElement>
  ) => {
    if (isLocked) {
      handleContainerTap()
      return
    }

    const rect = playerRef.current
      ?.getBoundingClientRect()
    if (!rect) return

    const tapX = e.clientX - rect.left
    const widthPercent = (tapX / rect.width) * 100
    const now = Date.now()
    const isDoubleTap = 
      now - lastTapRef.current.time < 300

    if (isDoubleTap) {
      if (widthPercent <= 40) {
        seekRelative(-10)
        triggerSeekFeedback('left')
      } else if (widthPercent >= 60) {
        seekRelative(10)
        triggerSeekFeedback('right')
      }
    } else {
      // Single tap → toggle controls
      // (only fires if no second tap follows)
      setTimeout(() => {
        if (Date.now() - lastTapRef.current.time 
          >= 300) {
          handleContainerTap()
        }
      }, 300)
    }

    lastTapRef.current = { time: now, x: tapX }
  }

  const triggerSeekFeedback = (
    side: 'left' | 'right'
  ) => {
    setSeekFeedback({ side, show: true })
    setTimeout(() => 
      setSeekFeedback(s => ({ ...s, show: false })), 
      600)
  }

  const seekRelative = (seconds: number) => {
    // Real postMessage to YouTube iframe API
    const iframe = document.getElementById(
      'yt-player') as HTMLIFrameElement
    iframe?.contentWindow?.postMessage(
      JSON.stringify({
        event: 'command',
        func: 'seekTo',
        args: [seconds, true],
      }), '*')
  }

  // ── LANDSCAPE TOGGLE ──
  const toggleLandscape = async () => {
    try {
      if (!isLandscape) {
        await ScreenOrientation.lock({
          orientation: 'landscape' })
        setIsLandscape(true)
      } else {
        await ScreenOrientation.lock({
          orientation: 'portrait' })
        setIsLandscape(false)
      }
    } catch {
      // Web fallback — just toggle visual state
      setIsLandscape(!isLandscape)
    }
  }

  useEffect(() => {
    return () => {
      ScreenOrientation.unlock().catch(() => {})
      if (hideTimerRef.current) 
        clearTimeout(hideTimerRef.current)
    }
  }, [])

  // ── AUDIO-ONLY MODE — screen off, audio plays ──
  const handleAudioOnlyToggle = () => {
    const next = !isAudioOnly
    setIsAudioOnly(next)
    onAudioOnlyToggle(next)
  }

  const sizeModeClass = {
    fit: 'object-contain',
    stretch: 'object-fill',
    zoom: 'object-cover',
  }[sizeMode]

  return (
    <div
      ref={playerRef}
      className={`relative bg-black overflow-hidden
        ${isLandscape 
          ? 'fixed inset-0 z-[70]' 
          : 'aspect-video w-full'}
        ${isAudioOnly ? 'hidden' : ''}`}
      onClick={handleZoneTap}
    >
      {/* YouTube iframe */}
      <iframe
        id="yt-player"
        src={`https://www.youtube.com/embed/` +
          `${youtubeId}?autoplay=1&controls=0` +
          `&modestbranding=1&rel=0&playsinline=1` +
          `&enablejsapi=1`}
        className={`w-full h-full border-0 ` +
          `${sizeModeClass}`}
        allow="autoplay; encrypted-media"
        title={title}
      />

      {/* LOCK OVERLAY — blocks all interaction */}
      {isLocked && (
        <div className="absolute inset-0 z-20"
          onClick={(e) => {
            e.stopPropagation()
            handleContainerTap()
          }}>
          {showLockHint && (
            <div className="absolute top-1/2 
              left-1/2 -translate-x-1/2 
              -translate-y-1/2 bg-black/70
              rounded-2xl px-5 py-3 flex 
              items-center gap-2">
              <svg width="18" height="18"
                viewBox="0 0 24 24" fill="none"
                stroke="white" strokeWidth="2">
                <rect x="3" y="11" width="18"
                  height="11" rx="2"/>
                <path d="M7 11V7a5 5 0 0 1 10 0v4"/>
              </svg>
              <span className="text-white text-sm">
                Screen Locked
              </span>
            </div>
          )}
          {/* Unlock button always visible */}
          <button
            onMouseDown={handleLockIconPressStart}
            onMouseUp={handleLockIconPressEnd}
            onTouchStart={handleLockIconPressStart}
            onTouchEnd={handleLockIconPressEnd}
            onClick={(e) => e.stopPropagation()}
            className="absolute bottom-6 right-6
              w-12 h-12 rounded-full bg-black/60
              flex items-center justify-center
              active:scale-90 transition-transform"
          >
            <svg width="20" height="20"
              viewBox="0 0 24 24" fill="none"
              stroke="white" strokeWidth="2">
              <rect x="3" y="11" width="18"
                height="11" rx="2"/>
              <path d="M7 11V7a5 5 0 0 1 10 0v4"/>
            </svg>
          </button>
        </div>
      )}

      {/* Seek feedback ripples */}
      {seekFeedback.show && (
        <div className={`absolute top-0 bottom-0
          ${seekFeedback.side === 'left' 
            ? 'left-0 w-2/5' : 'right-0 w-2/5'}

// Real implementation: when audio-only is on,
// the video <iframe> stays hidden (display:none
// is wrong since YouTube pauses hidden iframes
// on some Android WebViews) — instead we move
// it off-screen but keep it rendered + playing,
// and show a "Now Playing" mini bar with screen-
// off capability via Capacitor's Keep Awake OFF
// (so screen CAN turn off while audio continues,
// matching PlayIt's real behavior)

import { KeepAwake } from '@capacitor-community/keep-awake'

export const enableAudioOnlyMode = async (
  movieTitle: string
) => {
  // Allow screen to turn off (don't keep awake)
  try {
    await KeepAwake.allowSleep()
  } catch {}

  // Real background audio requires the iframe to
  // stay mounted off-screen, not removed from DOM
  return {
    containerStyle: {
      position: 'fixed' as const,
      top: '-9999px',
      left: '-9999px',
      width: '1px',
      height: '1px',
      opacity: 0,
      pointerEvents: 'none' as const,
    }
  }
}

export const disableAudioOnlyMode = async () => {
  try {
    await KeepAwake.keepAwake()
  } catch {}
}

'use client'
import { useState, useEffect } from 'react'
import { supabase } from '@/lib/supabase'

interface ActionBarProps {
  movieId: string
  movieTitle: string
  youtubeId: string
  authUserId: string | null
  isAudioOnly: boolean
  onToggleAudioOnly: () => void
}

export default function ActionBar({
  movieId, movieTitle, youtubeId, authUserId,
  isAudioOnly, onToggleAudioOnly
}: ActionBarProps) {
  const [reaction, setReaction] = useState<
    'like' | 'dislike' | null>(null)
  const [isSaved, setIsSaved] = useState(false)
  const [isLoading, setIsLoading] = useState(true)

  // Load REAL current state from Supabase
  useEffect(() => {
    if (!authUserId) {
      setIsLoading(false)
      return
    }

    const loadState = async () => {
      const [{ data: reactionData }, 
        { data: watchlistData }] = await Promise.all([
        supabase
          .from('movie_reactions')
          .select('reaction')
          .eq('user_id', authUserId)
          .eq('video_id', movieId)
          .maybeSingle(),
        supabase
          .from('user_watchlist')
          .select('id')
          .eq('user_id', authUserId)
          .eq('movie_id', movieId)
          .maybeSingle(),
      ])

      setReaction(
        (reactionData?.reaction as any) || null)
      setIsSaved(!!watchlistData)
      setIsLoading(false)
    }

    loadState()
  }, [movieId, authUserId])

  const handleReaction = async (
    type: 'like' | 'dislike'
  ) => {
    if (!authUserId) return

    const isRemoving = reaction === type
    const next = isRemoving ? null : type

    // Optimistic update
    setReaction(next)

    if (isRemoving) {
      await supabase
        .from('movie_reactions')
        .delete()
        .eq('user_id', authUserId)
        .eq('video_id', movieId)
    } else {
      await supabase
        .from('movie_reactions')
        .upsert([{
          user_id: authUserId,
          video_id: movieId,
          reaction: type,
        }], { onConflict: 'user_id,video_id' })
    }
  }

  const handleSave = async () => {
    if (!authUserId) return
    const next = !isSaved
    setIsSaved(next)

    if (next) {
      await supabase.from('user_watchlist').upsert([{
        user_id: authUserId,
        movie_id: movieId,
      }], { onConflict: 'user_id,movie_id' })
    } else {
      await supabase
        .from('user_watchlist')
        .delete()
        .eq('user_id', authUserId)
        .eq('movie_id', movieId)
    }
  }

  const handleShare = async () => {
    const shareUrl = 
      `${window.location.origin}/watch/${movieId}`
    const shareData = {
      title: movieTitle,
      url: shareUrl,
    }
    try {

import { supabase } from './supabase'

export interface RecommendedMovie {
  id: string
  title: string
  thumbnail: string
  channel_name: string
  channel_id: string
  genre?: string[]
}

export const getSuggestedMovies = async (
  currentMovieId: string,
  currentGenre: string[],
  authUserId: string | null
): Promise<RecommendedMovie[]> => {
  
  // ── 50% — Same genre as currently playing ──
  let genreMatches: RecommendedMovie[] = []
  if (currentGenre?.length) {
    const { data } = await supabase
      .from('movies')
      .select('id, title, thumbnail, channel_name,
        channel_id, genre')
      .neq('id', currentMovieId)
      .overlaps('genre', currentGenre)
      .eq('is_hidden', false)
      .limit(20)

    genreMatches = shuffleArray(data || []).slice(0, 5)
  }

  // ── 50% — User's top historical genre ──
  let historyMatches: RecommendedMovie[] = []

  if (authUserId) {
    // Real query: get user's watch history genres
    const { data: historyRows } = await supabase
      .from('user_history')
      .select('movie_id, movies(genre)')
      .eq('user_id', authUserId)
      .order('watched_at', { ascending: false })
      .limit(20)

    const genreCounts: Record<string, number> = {}
    for (const row of historyRows || []) {
      const genres = (row as any).movies?.genre || []
      for (const g of genres) {
        genreCounts[g] = (genreCounts[g] || 0) + 1
      }
    }

    const topGenre = Object.entries(genreCounts)
      .sort((a, b) => b[1] - a[1])[0]?.[0]

    if (topGenre) {
      const { data } = await supabase
        .from('movies')
        .select('id, title, thumbnail, channel_name,
          channel_id, genre')
        .neq('id', currentMovieId)
        .contains('genre', [topGenre])
        .eq('is_hidden', false)
        .order('like_count', { ascending: false })
        .limit(10)

      historyMatches = shuffleArray(data || [])
        .slice(0, 5)
    }
  }

  // ── REAL fallback for new users with no history ──
  // Trending = most liked in last 7 days (real query,
  // not fake "trending" data)
  if (historyMatches.length === 0) {
    const sevenDaysAgo = new Date(
      Date.now() - 7 * 86400000).toISOString()

    const { data: trendingReactions } = await supabase
      .from('movie_reactions')
      .select('video_id')
      .eq('reaction', 'like')
      .gte('created_at', sevenDaysAgo)

    const videoIds = [...new Set(
      (trendingReactions || []).map(r => r.video_id)
    )]

    if (videoIds.length > 0) {
      const { data } = await supabase
        .from('movies')
        .select('id, title, thumbnail, channel_name,
          channel_id, genre')
        .in('id', videoIds)
        .neq('id', currentMovieId)
        .limit(10)

      historyMatches = shuffleArray(data || [])
        .slice(0, 5)
    } else {
      // Real fallback if even that's empty:
      // most recently added movies (real data,
      // not fake placeholder)
      const { data } = await supabase
        .from('movies')
        .select('id, title, thumbnail, channel_name,
          channel_id, genre')
        .neq('id', currentMovieId)
        .order('created_at', { ascending: false })
        .limit(5)
      historyMatches = data || []
    }
  }

  // ── Merge, dedupe, shuffle ──
  const combined = [...genreMatches, ...historyMatches]
  const unique = Array.from(
    new Map(combined.map(m => [m.id, m])).values()
  )
  return shuffleArray(unique).slice(0, 10)
}

function shuffleArray<T>(arr: T[]): T[] {
  const a = [...arr]
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1))
    [a[i], a[j]] = [a[j], a[i]]
  }
  return a
}
