    63→function timeAgo(dateStr: string | null): string {
    64→  if (!dateStr) return 'Never'
    65→  const diff = Date.now() - new Date(dateStr).getTime()
    66→  const mins = Math.floor(diff / 60000)
    67→  if (mins < 1) return 'Just now'
    68→  if (mins < 60) return `${mins}m ago`
    69→  const hours = Math.floor(mins / 60)
    70→  if (hours < 24) return `${hours}h ago`
    71→  const days = Math.floor(hours / 24)
    72→  if (days < 30) return `${days}d ago`
    73→  const months = Math.floor(days / 30)
    74→  return `${months}mo ago`
    75→}

    77→function extractChannelId(url: string): string {
    78→  const patterns = [
    79→    /youtube\.com\/channel\/(UC[\w-]{22})/,
    80→    /youtube\.com\/@([\w.-]+)/,
    81→    /youtube\.com\/c\/([\w.-]+)/,
    82→    /youtube\.com\/user\/([\w.-]+)/,
    83→  ]
    84→  for (const p of patterns) {
    85→    const m = url.match(p)
    86→    if (m) return m[1]
    87→  }
    88→  if (/^UC[\w-]{22}$/.test(url.trim())) return url.trim()
    89→  return ''
    90→}
