   668→CREATE OR REPLACE FUNCTION deduct_user_coins(p_auth_uid UUID, p_amount INTEGER)
   669→RETURNS BOOLEAN LANGUAGE plpgsql SECURITY DEFINER AS $$
   670→DECLARE v_profile_id UUID; v_current_coins INTEGER;
   671→BEGIN
   672→  SELECT id INTO v_profile_id FROM user_profiles WHERE auth_user_id = p_auth_uid;
   673→  IF v_profile_id IS NULL THEN RETURN false; END IF;
   674→  SELECT coins INTO v_current_coins FROM user_profiles WHERE id = v_profile_id;
   675→  IF v_current_coins < p_amount THEN RETURN false; END IF;
   676→  UPDATE user_profiles SET coins = coins - p_amount WHERE id = v_profile_id;
   677→  RETURN true;
   678→END; $$;
