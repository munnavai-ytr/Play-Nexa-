    63→function timeAgo(dateStr: string | null): string {
    64→  if (!dateStr) return 'Never'
    65→  const diff = Date.now() - new Date(dateStr).getTime()
    66→  const mins = Math.floor(diff / 60000)
    67→  if (mins < 1) return 'Just now'
    68→  if (mins < 60) return `${mins}m ago`
    69→  const hours = Math.floor(mins / 60)
    70→  if (hours < 24) return `${hours}h ago`
    71→  const days = Math.floor(hours / 24)
    72→  if (days < 30) return `${days}d ago`
    73→  const months = Math.floor(days / 30)
    74→  return `${months}mo ago`
    75→}

    77→// Remove tracking params (?si=, etc.) from YouTube URLs
    78→const cleanYouTubeUrl = (url: string): string => {
    79→  try {
    80→    // Remove all query params (?si=, ?v=, etc)
    81→    const urlObj = new URL(url.trim())
    82→    // Keep only origin + pathname
    83→    const clean = urlObj.origin + urlObj.pathname
    84→    return clean
    85→  } catch {
    86→    // If not valid URL, return trimmed
    87→    return url.trim()
    88→  }
    89→}
    90→
    91→// Extract channel identifier (handle, UC ID, user, c-name) from URL
    92→const extractChannelIdentifier = (url: string): string => {
    93→  const cleaned = cleanYouTubeUrl(url)
    94→
    95→  // Match patterns:
    96→  // youtube.com/channel/UCxxxxxx
    97→  const ucMatch = cleaned.match(/channel\/(UC[\w-]{10,})/)
    98→  if (ucMatch) return ucMatch[1]
    99→
   100→  // youtube.com/@handle
   101→  const handleMatch = cleaned.match(/\/@([\w.-]+)/)
   102→  if (handleMatch) return '@' + handleMatch[1]
   103→
   104→  // youtube.com/c/name
   105→  const cMatch = cleaned.match(/\/c\/([\w.-]+)/)
   106→  if (cMatch) return cMatch[1]
   107→
   108→  // youtube.com/user/name
   109→  const userMatch = cleaned.match(/\/user\/([\w.-]+)/)
   110→  if (userMatch) return userMatch[1]
   111→
   112→  // Bare UC.. ID pasted directly
   113→  if (/^UC[\w-]{10,}$/.test(cleaned.trim())) return cleaned.trim()
   114→  // Bare @handle pasted directly
   115→  if (/^@[\w.-]+$/.test(cleaned.trim())) return cleaned.trim()
   116→
   117→  return cleaned
   118→}
