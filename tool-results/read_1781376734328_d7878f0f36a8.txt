    75→function timeAgo(dateStr: string | null): string {
    76→  if (!dateStr) return 'Never'
    77→  const diff = Date.now() - new Date(dateStr).getTime()
    78→  const mins = Math.floor(diff / 60000)
    79→  if (mins < 1) return 'Just now'
    80→  if (mins < 60) return `${mins}m ago`
    81→  const hours = Math.floor(mins / 60)
    82→  if (hours < 24) return `${hours}h ago`
    83→  const days = Math.floor(hours / 24)
    84→  if (days < 30) return `${days}d ago`
    85→  const months = Math.floor(days / 30)
    86→  return `${months}mo ago`
    87→}

    89→// Remove tracking params (?si=, etc.) from YouTube URLs
    90→const cleanYouTubeUrl = (url: string): string => {
    91→  try {
    92→    // Remove all query params (?si=, ?v=, etc)
    93→    const urlObj = new URL(url.trim())
    94→    // Keep only origin + pathname
    95→    const clean = urlObj.origin + urlObj.pathname
    96→    return clean
    97→  } catch {
    98→    // If not valid URL, return trimmed
    99→    return url.trim()
   100→  }
   101→}
   102→
   103→// Extract channel identifier (handle, UC ID, user, c-name) from URL
   104→const extractChannelIdentifier = (url: string): string => {
   105→  const cleaned = cleanYouTubeUrl(url)
   106→
   107→  // Match patterns:
   108→  // youtube.com/channel/UCxxxxxx
   109→  const ucMatch = cleaned.match(/channel\/(UC[\w-]{10,})/)
   110→  if (ucMatch) return ucMatch[1]
   111→
   112→  // youtube.com/@handle
   113→  const handleMatch = cleaned.match(/\/@([\w.-]+)/)
   114→  if (handleMatch) return '@' + handleMatch[1]
   115→
   116→  // youtube.com/c/name
   117→  const cMatch = cleaned.match(/\/c\/([\w.-]+)/)
   118→  if (cMatch) return cMatch[1]
   119→
   120→  // youtube.com/user/name
   121→  const userMatch = cleaned.match(/\/user\/([\w.-]+)/)
   122→  if (userMatch) return userMatch[1]
   123→
   124→  // Bare UC.. ID pasted directly
   125→  if (/^UC[\w-]{10,}$/.test(cleaned.trim())) return cleaned.trim()
   126→  // Bare @handle pasted directly
   127→  if (/^@[\w.-]+$/.test(cleaned.trim())) return cleaned.trim()
   128→
   129→  return cleaned
   130→}
